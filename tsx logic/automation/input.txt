```tsx
import React from 'react';
import {
  useCurrentFrame,
  useVideoConfig,
  interpolate,
  spring,
  AbsoluteFill,
} from 'remotion';

export const compositionConfig = {
  id: 'Scene1Ch5',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Scene1Ch5: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({
      frame: Math.max(0, frame - delay),
      fps,
      config: { damping: 12, mass: 0.4, stiffness: 120, ...config },
    });

  const exitOp = interpolate(
    frame,
    [durationInFrames - 15, durationInFrames],
    [1, 0],
    { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
  );

  // Diagonal split reveal
  const splitReveal = interpolate(frame, [0, 40], [0, 1], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  // Scale pulse for main word
  const pulseScale = 1 + Math.sin(frame * 0.08) * 0.015;

  // Scan line animation
  const scanY = interpolate(frame, [0, durationInFrames], [-100, 1200], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  // Warning flash
  const warningOp = interpolate(
    frame,
    [50, 55, 60, 65],
    [0, 1, 0.4, 1],
    { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
  );

  return (
    <AbsoluteFill
      style={{
        backgroundColor: '#1C1C1C',
        overflow: 'hidden',
        opacity: exitOp,
      }}
    >
      {/* Gradient — deep red-to-black diagonal */}
      <AbsoluteFill
        style={{
          background:
            'linear-gradient(135deg, #1C1C1C 0%, #2A0A0A 40%, #1C1C1C 100%)',
        }}
      />

      {/* Geometric shape layer — large tilted rectangle */}
      <AbsoluteFill>
        <svg width="1920" height="1080" style={{ position: 'absolute' }}>
          <rect
            x="-200"
            y="200"
            width="900"
            height="900"
            fill="none"
            stroke="#ff0000"
            strokeWidth="1"
            opacity={0.04}
            transform={`rotate(${frame * 0.05} 960 540)`}
          />
          <rect
            x="1100"
            y="-100"
            width="700"
            height="700"
            fill="none"
            stroke="#e9d700"
            strokeWidth="1"
            opacity={0.04}
            transform={`rotate(${-frame * 0.03} 960 540)`}
          />
          {/* Large circle */}
          <circle
            cx="960"
            cy="540"
            r="500"
            fill="none"
            stroke="#F5F5F5"
            strokeWidth="0.5"
            opacity={0.03}
          />
        </svg>
      </AbsoluteFill>

      {/* Motion texture — horizontal scan lines */}
      <AbsoluteFill>
        <svg width="1920" height="1080" style={{ position: 'absolute' }}>
          {Array.from({ length: 20 }).map((_, i) => (
            <line
              key={i}
              x1="0"
              y1={i * 60}
              x2="1920"
              y2={i * 60}
              stroke="#F5F5F5"
              strokeWidth="0.3"
              opacity={0.04}
            />
          ))}
          {/* Scan line */}
          <line
            x1="0"
            y1={scanY}
            x2="1920"
            y2={scanY}
            stroke="#ff0000"
            strokeWidth="2"
            opacity={0.15}
          />
        </svg>
      </AbsoluteFill>

      {/* Floating particles */}
      <AbsoluteFill>
        <svg width="1920" height="1080" style={{ position: 'absolute' }}>
          {Array.from({ length: 30 }).map((_, i) => {
            const cx = seededRandom(i * 3) * 1920;
            const cy = seededRandom(i * 3 + 1) * 1080;
            const r = seededRandom(i * 3 + 2) * 3 + 1;
            const drift = Math.sin(frame * 0.04 + i) * 12;
            return (
              <circle
                key={i}
                cx={cx + drift}
                cy={cy - (frame * (0.3 + seededRandom(i) * 0.4)) % 1080}
                r={r}
                fill={i % 3 === 0 ? '#e9d700' : i % 3 === 1 ? '#ff0000' : '#F5F5F5'}
                opacity={0.06 + seededRandom(i * 7) * 0.08}
              />
            );
          })}
        </svg>
      </AbsoluteFill>

      {/* Main content */}
      <AbsoluteFill
        style={{
          display: 'flex',
          flexDirection: 'column',
          justifyContent: 'center',
          alignItems: 'center',
          padding: '0 120px',
        }}
      >
        {/* Top label */}
        <div
          style={{
            fontFamily: 'JetBrains Mono',
            fontSize: 28,
            color: '#ff0000',
            letterSpacing: 8,
            opacity: warningOp,
            marginBottom: 40,
            textTransform: 'uppercase',
          }}
        >
          ⚠ KAUM JEMAND SPRICHT DARÜBER
        </div>

        {/* Main headline */}
        <div
          style={{
            fontFamily: 'Anton',
            fontSize: 200,
            color: '#F5F5F5',
            lineHeight: 0.85,
            textAlign: 'center',
            transform: `scale(${sp(10) * pulseScale})`,
            opacity: sp(10),
            letterSpacing: -4,
          }}
        >
          STEUER
          <br />
          <span style={{ color: '#e9d700' }}>ASYMMETRIE</span>
        </div>

        {/* Sub text */}
        <div
          style={{
            fontFamily: 'Open Sans',
            fontSize: 48,
            color: '#9E9E9E',
            marginTop: 40,
            opacity: sp(35),
            transform: `translateY(${(1 - sp(35)) * 30}px)`,
            textAlign: 'center',
            maxWidth: 900,
          }}
        >
          Das Finanzamt behandelt nicht alle Einkommensarten gleich
        </div>

        {/* Red underline bar */}
        <div
          style={{
            width: `${splitReveal * 600}px`,
            height: 4,
            backgroundColor: '#ff0000',
            marginTop: 30,
            borderRadius: 2,
          }}
        />
      </AbsoluteFill>

      {/* Stick figure */}
      <AbsoluteFill>
        <div
          style={{
            position: 'absolute',
            bottom: 140,
            left: 100,
            opacity: sp(50),
            transform: `translateX(${(1 - sp(50)) * -60}px)`,
          }}
        >
          <svg
            width="90"
            height="160"
            style={{
              transform: `translateY(${Math.sin(frame * 0.05) * 2}px)`,
            }}
          >
            {/* Golden glow */}
            <circle cx="45" cy="30" r="28" fill="#e9d700" opacity={0.12} />
            {/* Head */}
            <circle
              cx="45"
              cy="30"
              r="18"
              fill="none"
              stroke="#F5F5F5"
              strokeWidth="2.5"
              style={{
                transform: `rotate(${Math.sin(frame * 0.04) * 3}deg)`,
                transformOrigin: '45px 30px',
              }}
            />
            {/* Glasses */}
            <circle
              cx="39"
              cy="30"
              r="5"
              fill="none"
              stroke="#e9d700"
              strokeWidth="1.8"
            />
            <circle
              cx="51"
              cy="30"
              r="5"
              fill="none"
              stroke="#e9d700"
              strokeWidth="1.8"
            />
            <line
              x1="44"
              y1="30"
              x2="46"
              y2="30"
              stroke="#e9d700"
              strokeWidth="1.5"
            />
            {/* Body */}
            <line
              x1="45"
              y1="48"
              x2="45"
              y2="110"
              stroke="#F5F5F5"
              strokeWidth="2.5"
            />
            {/* Arms — raised alert */}
            <line
              x1="45"
              y1="65"
              x2="15"
              y2="50"
              stroke="#F5F5F5"
              strokeWidth="2.5"
            />
            <line
              x1="45"
              y1="65"
              x2="75"
              y2="50"
              stroke="#F5F5F5"
              strokeWidth="2.5"
            />
            {/* Legs */}
```
