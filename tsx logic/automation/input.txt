  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

  const aurScale = 1 + Math.sin(frame * 0.15) * 0.08;
  const aurGlow = Math.sin(frame * 0.12) * 0.5 + 0.5;

  // Wrong game badge shake
  const badgeRot = frame > 60 ? Math.sin(frame * 0.3) * 2 : 0;

  return (
    <AbsoluteFill
      style={{
        backgroundColor: '#1C1C1C',
        overflow: 'hidden',
        opacity: exitOp,

    >
      {/* Animated gradient — deep red danger pulse */}

            position: 'absolute',

          }}
        />
        <div
          style={{
            position: 'absolute',
            inset: 0,
            background: `radial-gradient(ellipse 50% 40% at 80% 70%, #1a1a2e 0%, transparent 70%)`,
          }}
        />
      </AbsoluteFill>

      {/* Scanline texture */}
      <AbsoluteFill style={{ opacity: 0.06 }}>
        {Array.from({ length: 30 }).map((_, i) => (
          <div
            key={i}
            style={{
              position: 'absolute',
              left: 0,
              right: 0,
              height: 2,
              top: ((i * 36 + scanY) % 1080),
              backgroundColor: '#F5F5F5',
            }}
          />
        ))}
      </AbsoluteFill>

      {/* Shape layers — slow drifting rectangles */}
      <AbsoluteFill style={{ opacity: 0.05 }}>
        {[
          { x: 100, y: 200, w: 400, h: 3, rot: 15, seed: 1 },
          { x: 1400, y: 100, w: 300, h: 3, rot: -20, seed: 2 },
          { x: 600, y: 800, w: 500, h: 3, rot: 8, seed: 3 },
          { x: 200, y: 600, w: 200, h: 200, rot: 30, seed: 4 },
          { x: 1600, y: 500, w: 180, h: 180, rot: -45, seed: 5 },
        ].map((s, i) => (
          <div
            key={i}
            style={{
              position: 'absolute',
              left: s.x + Math.sin(frame * 0.01 + s.seed) * 15,
              top: s.y + Math.cos(frame * 0.008 + s.seed) * 10,
              width: s.w,
              height: s.h,
              border: '2px solid #ff0000',
              transform: `rotate(${s.rot + frame * 0.02}deg)`,
            }}
          />
        ))}
      </AbsoluteFill>

      {/* Floating particles */}
      <AbsoluteFill>
        {Array.from({ length: 18 }).map((_, i) => {
