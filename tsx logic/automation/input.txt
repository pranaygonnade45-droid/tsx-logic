import React from 'react';
import { useCurrentFrame, useVideoConfig, interpolate, Easing, spring, AbsoluteFill } from 'remotion';

export const compositionConfig = {
  id: 'Scene1_Hebel',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Scene1_Hebel: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Shockwave ring expand
  const shockwave = interpolate(frame, [0, 40], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp', easing: Easing.out(Easing.cubic) });
  const shockOp = interpolate(frame, [0, 40], [0.6, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Camera zoom
  const camScale = interpolate(frame, [0, 60], [1.08, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp', easing: Easing.out(Easing.quad) });

  // Text split reveal
  const word1Op = sp(10);
  const word2Op = sp(25);
  const word3Y = interpolate(sp(40), [0, 1], [60, 0]);
  const word3Op = sp(40);

  // Lever SVG animate
  const leverRotate = interpolate(sp(50), [0, 1], [-30, 0]);
  const leverScale = sp(50, { stiffness: 200, damping: 10 });

  // Line draw
  const lineDraw = interpolate(frame, [55, 90], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Particles
  const particles = Array.from({ length: 18 }, (_, i) => ({
    x: seededRandom(i * 3) * 1920,
    y: seededRandom(i * 3 + 1) * 1080,
    size: seededRandom(i * 3 + 2) * 5 + 2,
    speed: seededRandom(i * 7) * 0.6 + 0.2,
    phase: seededRandom(i * 11) * Math.PI * 2,
  }));

  // Screen shake on impact
  const shakeX = frame < 8 ? Math.sin(frame * 15) * 6 * Math.max(0, 1 - frame / 8) : 0;
  const shakeY = frame < 8 ? Math.cos(frame * 13) * 4 * Math.max(0, 1 - frame / 8) : 0;

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>
      {/* Gradient BG */}
      <AbsoluteFill style={{ transform: `scale(${camScale}) translate(${shakeX}px, ${shakeY}px)` }}>
        <AbsoluteFill style={{
          background: 'radial-gradient(ellipse at 20% 50%, #2a1f00 0%, #1C1C1C 55%), linear-gradient(135deg, #1C1C1C 0%, #2D2D2D 100%)',
        }} />

        {/* Shockwave ring */}
        <AbsoluteFill style={{ display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
          <div style={{
            width: shockwave * 1400,
            height: shockwave * 1400,
            borderRadius: '50%',
            border: `4px solid #e9d700`,
            opacity: shockOp,
            position: 'absolute',
          }} />
        </AbsoluteFill>

        {/* Shape layer — large triangle */}
        <svg style={{ position: 'absolute', top: -80, right: -60, opacity: 0.05 }} width="600" height="600">
          <polygon points="300,0 600,600 0,600"
            fill="none" stroke="#e9d700" strokeWidth="3"
            transform={`rotate(${frame * 0.08} 300 300)`} />
        </svg>

        {/* Shape layer — circle drift */}
        <div style={{
          position: 'absolute', left: -120, bottom: -80,
          width: 500, height: 500, borderRadius: '50%',
          border: '2px solid #F5F5F5', opacity: 0.04,
          transform: `translate(${Math.sin(frame * 0.025) * 20}px, ${Math.cos(frame * 0.02) * 15}px)`,
        }} />

        {/* Motion texture — grid lines */}
        <svg style={{ position: 'absolute', inset: 0, opacity: 0.04 }} width="1920" height="1080">
          {Array.from({ length: 20 }, (_, i) => (
            <line key={i} x1={i * 100} y1="0" x2={i * 100} y2="1080" stroke="#e9d700" strokeWidth="1" />
          ))}
          {Array.from({ length: 12 }, (_, i) => (
            <line key={i + 20} x1="0" y1={i * 90} x2="1920" y2={i * 90} stroke="#e9d700" strokeWidth="1" />
          ))}
        </svg>

        {/* Floating particles */}
        {particles.map((p, i) => (
          <div key={i} style={{
            position: 'absolute',
            left: p.x + Math.sin(frame * p.speed + p.phase) * 25,
            top: p.y + Math.cos(frame * p.speed * 0.7 + p.phase) * 18,
            width: p.size, height: p.size,
            borderRadius: '50%',
            backgroundColor: i % 3 === 0 ? '#e9d700' : '#F5F5F5',
            opacity: 0.25 + Math.sin(frame * 0.04 + p.phase) * 0.1,
          }} />
        ))}

        {/* MAIN — Divider line animated */}
        <div style={{
          position: 'absolute', left: 160, top: '50%',
          width: lineDraw * 640, height: 3,
          background: 'linear-gradient(90deg, #e9d700, transparent)',
          borderRadius: 2,
        }} />

        {/* LEFT — "EMPÖRUNG" CROSSED OUT */}
        <div style={{
          position: 'absolute', left: 160, top: 320,
          opacity: word1Op,
          transform: `translateX(${interpolate(word1Op, [0, 1], [-80, 0])}px)`,
        }}>
          <div style={{ fontFamily: 'Anton', fontSize: 72, color: '#9E9E9E', letterSpacing: 4, position: 'relative' }}>
            EMPÖRUNG
            {/* Cross out */}
            <div style={{
              position: 'absolute', top: '50%', left: 0,
              width: `${lineDraw * 100}%`, height: 4,
              backgroundColor: '#ff0000', borderRadius: 2,
            }} />
          </div>
          <div style={{ fontFamily: 'Open Sans', fontSize: 28, color: '#9E9E9E', marginTop: 8 }}>
            Macht dich müde
          </div>
        </div>

        {/* Arrow between */}
        <div style={{
          position: 'absolute', left: 730, top: 340,
          opacity: word2Op,
          transform: `scale(${word2Op})`,
        }}>
          <svg width="80" height="60" viewBox="0 0 80 60">
            <path d="M10 30 L60 30 M45 15 L65 30 L45 45" stroke="#e9d700" strokeWidth="4" fill="none" strokeLinecap="round" strokeLinejoin="round" />
          </svg>
        </div>

        {/* RIGHT — "STRUKTUR" glowing */}
        <div style={{
          position: 'absolute', left: 840, top: 280,
          opacity: word3Op,
          transform: `translateY(${word3Y}px)`,
        }}>
          <div style={{
            fontFamily: 'Anton', fontSize: 130, color: '#e9d700',
            letterSpacing: 6, lineHeight: 1,
            textShadow: `0 0 ${sp(40) * 40}px rgba(233,215,0,0.6), 0 0 ${sp(40) * 80}px rgba(233,215,0,0.3)`,
          }}>
            STRUKTUR
          </div>
          <div style={{ fontFamily: 'Open Sans', fontSize: 32, color: '#F5F5F5', marginTop: 12, opacity: 0.85 }}>
            Bringt dich vorwärts
          </div>
        </div>

        {/* LEVER SVG — bottom center */}
        <div style={{
          position: 'absolute', left: '50%', bottom: 80,
          transform: `translateX(-50%) rotate(${leverRotate}deg) scale(${leverScale})`,
          transformOrigin: 'bottom center',
        }}>
          <svg width="120" height="140" viewBox="0 0 120 140">
            {/* Base */}
            <ellipse cx="60" cy="130" rx="40" ry="10" fill="#2D2D2D" stroke="#e9d700" strokeWidth="2" />
            {/* Pole */}
            <rect x="55" y="30" width="10" height="100" rx="5" fill="#9E9E9E" />
            {/* Ball */}
            <circle cx="60" cy="25" r="20" fill="#e9d700" />
            <circle cx="52" cy="18" r="6" fill="#FFC107" opacity="0.7" />
          </svg>
        </div>

        {/* BOTTOM LABEL */}
        <div style={{
          position: 'absolute', bottom: 30, left: '50%',
          transform: 'translateX(-50%)',
          fontFamily: 'Inter', fontSize: 22, color: '#e9d700', letterSpacing: 8, fontWeight: 700,
          opacity: sp(70),
        }}>
          DEIN HEBEL BEGINNT HIER
        </div>

        {/* Stick figure */}
        <div style={{
          position: 'absolute', right: 140, bottom: 140,
          opacity: sp(55),
          transform: `scale(${sp(55)}) translateY(${Math.sin(frame * 0.05) * 3}px)`,
        }}>
          <svg width="70" height="130" viewBox="0 0 70 130">
            {/* Glow */}
            <circle cx="35" cy="25" r="28" fill="#e9d700" opacity="0.12" />
            {/* Head */}
            <circle cx="35" cy="22" r="16" fill="none" stroke="#F5F5F5" strokeWidth="3" />
            {/* Glasses */}
            <rect x="24" y="19" width="8" height="6" rx="2" fill="none" stroke="#e9d700" strokeWidth="2" />
            <rect x="37" y="19" width="8" height="6" rx="2" fill="none" stroke="#e9d700" strokeWidth="2" />
            <line x1="32" y1="22" x2="37" y2="22" stroke="#e9d700" strokeWidth="1.5" />
            {/* Body */}
            <line x1="35" y1="38" x2="35" y2="85" stroke="#F5F5F5" strokeWidth="3" strokeLinecap="round" />
            {/* Arms raised — celebrating */}
            <line x1="35" y1="55" x2="10" y2="35" stroke="#F5F5F5" strokeWidth="3" strokeLinecap="round" />
            <line x1="35" y1="55" x2="60" y2="35" stroke="#F5F5F5" strokeWidth="3" strokeLinecap="round" />

        {/* Vignette */}
        <AbsoluteFill style={{
