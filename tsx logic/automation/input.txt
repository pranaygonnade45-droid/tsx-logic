	useCurrentFrame,
	useVideoConfig,

	const x = Math.sin(seed * 9999) * 10000;

		extrapolateRight: 'clamp',

		extrapolateRight: 'clamp',
	});

		extrapolateRight: 'clamp',

				/>

				{/* Animated shape layers */}
				{[...Array(6)].map((_, i) => {
					const seed = seededRandom(i + 10);
					const size = 150 + seed * 350;
					const startX = seededRandom(i + 20) * 1920;
					const startY = seededRandom(i + 30) * 1080;
					const drift = interpolate(
						frame,
						[0, durationInFrames],
						[0, 40 + seed * 60],
						{ extrapolateRight: 'clamp' }
					);
					const rot = interpolate(
						frame,
						[0, durationInFrames],
						[0, 15 + seed * 30],
						{ extrapolateRight: 'clamp' }
					);
					const breathe =
						1 + Math.sin(frame * 0.02 + i) * 0.04;
					const shapes = ['circle', 'rect', 'triangle'];
					const shape = shapes[i % 3];

					return (
						<div
							key={`shape-${i}`}
							style={{
								position: 'absolute',
								left: startX,
								top: startY + drift * (i % 2 === 0 ? -1 : 1),
								width: size,
								height: size,
								transform: `rotate(${rot}deg) scale(${breathe})`,
								opacity: 0.04 + seed * 0.04,
							}}
						>
							{shape === 'circle' && (
								<svg width={size} height={size}>
									<circle
										cx={size / 2}
										cy={size / 2}
										r={size / 2 - 2}
										fill="none"
										stroke="#9E9E9E"
										strokeWidth="1.5"
									/>
								</svg>
							)}
							{shape === 'rect' && (
								<svg width={size} height={size}>
									<rect
										x="2"
										y="2"
										width={size - 4}
										height={size - 4}

										stroke="#FFC107"
										strokeWidth="1"
										rx="8"

							{shape === 'triangle' && (

									<polygon

										fill="none"
										stroke="#e9d700"
										strokeWidth="1"
									/>
								</svg>
							)}
						</div>
					);
				})}

				{/* Motion texture — diagonal lines */}
				<AbsoluteFill style={{ opacity: 0.03 }}>
					{[...Array(20)].map((_, i) => (
						<div
							key={`line-${i}`}
							style={{
								position: 'absolute',
								left: -200 + i * 120,
								top: 0,
								width: 1,
								height: 2000,

							}}
						/>
					))}
				</AbsoluteFill>

				{/* Floating particles */}
				{[...Array(30)].map((_, i) => {

					const pSize = 2 + seededRandom(i + 300) * 4;

						<div

					);

					const wy = 80 + seededRandom(i + 60) * 850;
					const wordFloat = Math.sin(frame * 0.025 + i * 1.3) * 8;
					const wordRot = Math.sin(frame * 0.015 + i * 2) * 4;
					const wordPulse =
						i === Math.floor((frame / 25) % floatingWords.length)
							? 1.3
							: 1;

					return (
						<div
							key={`word-${i}`}
							style={{
								position: 'absolute',
								left: wx,
								top: wy + wordFloat,
								transform: `scale(${wordSpring * wordPulse}) rotate(${wordRot}deg)`,
								opacity: wordSpring * 0.5,
								fontFamily: 'Noto Sans Devanagari',
								fontSize: 38 + seededRandom(i + 70) * 22,
								fontWeight: 700,
								color:
									i % 4 === 0
										? '#FFC107'
										: i % 4 === 1
											? '#e9d700'
											: i % 4 === 2
												? '#ff0000'
												: '#9E9E9E',
								letterSpacing: '2px',
								textShadow: '0 0 20px rgba(0,0,0,0.6)',
							}}
						>
							{word}
						</div>
					);
				})}

				{/* Center question text */}
				<div
					style={{
						position: 'absolute',
						top: '50%',
						left: '50%',
						transform: `translate(-50%, -50%) scale(${sp(20, { damping: 10, stiffness: 90 })})`,
						textAlign: 'center',
						width: 1400,
					}}
				>
					{/* Glow behind text */}
					<div
						style={{
							position: 'absolute',
							top: '50%',
							left: '50%',
							transform: 'translate(-50%, -50%)',

					<div
						style={{
							fontFamily: 'Yantra One',
							fontSize: 110,

						}}
					>
						क्या छोड़ नहीं पा रहे?
					</div>

					{/* Underline accent */}
					<div
						style={{
							margin: '30px auto 0',
							width: interpolate(sp(55), [0, 1], [0, 500]),
							height: 4,
							background:
								'linear-gradient(90deg, transparent, #FFC107, #e9d700, #FFC107, transparent)',
							borderRadius: 2,
						}}
					/>
				</div>

				{/* Bottom prompt */}
				<div
					style={{
						position: 'absolute',
						bottom: 100,
						left: '50%',
						transform: `translateX(-50%) translateY(${(1 - sp(70)) * 40}px)`,
						opacity: sp(70),
						fontFamily: 'Noto Sans Devanagari',
						fontSize: 36,
						fontWeight: 600,
						color: '#FFC107',
						textAlign: 'center',

					एक शब्द में लिख दीजिये ↓

				{/* Stick figure — thinking pose */}
				<div
					style={{
						position: 'absolute',
						right: 120,

					{/* Golden glow */}

							height: 200,
							borderRadius: '50%',
							background:
								'radial-gradient(circle, rgba(255,193,7,0.15) 0%, transparent 70%)',
							filter: 'blur(20px)',

					/>
					<svg
						width="120"

import React from 'react';
import {
	AbsoluteFill,
	useCurrentFrame,
	useVideoConfig,
	interpolate,
	spring,
	Easing,
} from 'remotion';

export const compositionConfig = {
	id: 'Chapter5Scene2_KahaaniMiltiJulti',
	durationInSeconds: 12,
	fps: 30,
	width: 1920,
	height: 1080,
};

const seededRandom = (seed: number): number => {
	const x = Math.sin(seed * 9999) * 10000;
	return x - Math.floor(x);
};

const Chapter5Scene2: React.FC = () => {
	const frame = useCurrentFrame();
	const { fps, durationInFrames } = useVideoConfig();

	const sp = (delay: number, config?: object) =>
		spring({
			frame: Math.max(0, frame - delay),

		});

	const exitOp = interpolate(
		frame,
		[durationInFrames - 15, durationInFrames],
		[1, 0],
		{ extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
	);

	// Camera tilt
	const cameraTilt = interpolate(frame, [0, durationInFrames], [0.5, -0.5], {
		extrapolateRight: 'clamp',
	});

	const gradShift = interpolate(frame, [0, durationInFrames], [200, 280], {
		extrapolateRight: 'clamp',
	});

	// Four category cards
	const categories = [
		{
			label: 'मकान',
			icon: 'house',
			color: '#FFC107',
			bg: 'rgba(255,193,7,0.08)',

		{
			label: 'गाड़ी',
			icon: 'car',
			color: '#e9d700',
			bg: 'rgba(233,215,0,0.08)',

		{

			label: 'कुछ और?',
			icon: 'question',

			bg: 'rgba(158,158,158,0.06)',
		},
	];

	return (
		<AbsoluteFill
			style={{
				backgroundColor: '#1C1C1C',
				overflow: 'hidden',
				opacity: exitOp,
			}}
		>
			<div
				style={{
					width: '100%',
					height: '100%',
					transform: `rotate(${cameraTilt}deg)`,
				}}
			>
				{/* Gradient background */}
				<AbsoluteFill
					style={{
						background: `linear-gradient(${gradShift}deg, #0d1117 0%, #1C1C1C 30%, #2a1a0a 60%, #1C1C1C 100%)`,
					}}
				/>

				{/* Shape layers — hexagons and circles */}
				{[...Array(7)].map((_, i) => {

					const size = 100 + seed * 400;

					const sy = seededRandom(i + 60) * 1080;
					const drift = Math.sin(frame * 0.012 + i) * 30;
					const rot = frame * 0.15 + i * 45;
					const breathe = 1 + Math.sin(frame * 0.018 + i * 1.2) * 0.03;

					return (
						<div
							key={`sh-${i}`}
							style={{
								position: 'absolute',
								left: sx,
								top: sy + drift,
								width: size,
								height: size,
								transform: `rotate(${rot}deg) scale(${breathe})`,
								opacity: 0.035,
							}}
						>
							<svg width={size} height={size}>
								{i % 2 === 0 ? (
									<polygon
										points={(() => {
											const r = size / 2 - 3;
											const cx = size / 2;
											const cy = size / 2;

												const angle =

												return `${cx + r * Math.cos(angle)},${cy + r * Math.sin(angle)}`;
											}).join(' ');
										})()}
										fill="none"
										stroke="#FFC107"
										strokeWidth="1.5"
									/>
								) : (
									<circle
										cx={size / 2}
										cy={size / 2}
										r={size / 2 - 3}
										fill="none"
										stroke="#e9d700"
										strokeWidth="1"
									/>
								)}
							</svg>
						</div>
					);
				})}

				{/* Motion texture — concentric rings */}
				<div
					style={{
						position: 'absolute',
						top: '50%',
						left: '50%',
						transform: 'translate(-50%, -50%)',
						opacity: 0.03,

					{[...Array(8)].map((_, i) => {
						const ringSize =
							200 +
							i * 120 +
							Math.sin(frame * 0.02 + i) * 15;
						return (
							<div
								key={`ring-${i}`}

									width: ringSize,

				{/* Title */}

						top: 70,
						left: '50%',
						transform: `translateX(-50%) scale(${sp(5)})`,
						fontFamily: 'Yantra One',
						fontSize: 72,
						fontWeight: 900,
						color: '#e9d700',
						textAlign: 'center',
						textShadow: '0 4px 25px rgba(0,0,0,0.6)',
					}}
				>
					कितनों की कहानी{' '}
					<span style={{ color: '#F5F5F5' }}>मिलती-जुलती</span>
				</div>

				{/* Four cards grid */}
				<div
					style={{
						position: 'absolute',
						top: '50%',
						left: '50%',
						transform: 'translate(-50%, -45%)',
						display: 'flex',
						gap: 50,
					}}
				>
					{categories.map((cat, i) => {
						const cardSpring = sp(25 + i * 18, {
							damping: 10,
							stiffness: 100,
						});
						const cardHover =
							Math.sin(frame * 0.04 + i * 1.5) * 5;
						const isActive =
							Math.floor(frame / 50) % 4 === i;
						const glowScale = isActive ? 1.08 : 1;
						const borderColor = isActive
							? cat.color
							: 'rgba(158,158,158,0.2)';

						return (
							<div
								key={`card-${i}`}
								style={{
									width: 320,
									height: 380,
									borderRadius: 24,
									background: cat.bg,
									border: `3px solid ${borderColor}`,
									display: 'flex',
									flexDirection: 'column',
									alignItems: 'center',
									justifyContent: 'center',
									gap: 25,
									transform: `scale(${cardSpring * glowScale}) translateY(${cardHover}px)`,
									opacity: cardSpring,
									transition: 'border-color 0.3s',
									boxShadow: isActive
										? `0 0 40px ${cat.color}33`
										: 'none',
								}}
							>
								{/* Icon */}
								<svg
									width="90"
									height="90"
									viewBox="0 0 90 90"
								>
									{cat.icon === 'house' && (
										<>
											<polygon
												points="45,10 80,45 70,45 70,80 20,80 20,45 10,45"
												fill="none"
												stroke={cat.color}
												strokeWidth="3.5"
												strokeLinejoin="round"
											/>
											<rect
												x="35"
												y="55"
												width="20"
												height="25"

												stroke={cat.color}
												strokeWidth="2.5"
