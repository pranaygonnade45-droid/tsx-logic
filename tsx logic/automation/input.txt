import React from 'react';
import {
  useCurrentFrame,
  useVideoConfig,
  interpolate,
  spring,
  AbsoluteFill,
  Easing,
} from 'remotion';

export const compositionConfig = {
  id: 'Chapter5Scene1',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Chapter5Scene1: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({
      frame: Math.max(0, frame - delay),
      fps,
      config: { damping: 12, mass: 0.4, stiffness: 120, ...config },
    });

  const exitOp = interpolate(
    frame,
    [durationInFrames - 15, durationInFrames],
    [1, 0],
    { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
  );

  // Camera zoom
  const cameraZoom = interpolate(frame, [0, durationInFrames], [1, 1.06], {
    extrapolateRight: 'clamp',
  });

  // Giant number animation
  const numberReveal = sp(15, { damping: 14, stiffness: 80 });
  const numberScale = interpolate(numberReveal, [0, 1], [3, 1]);
  const numberOp = interpolate(numberReveal, [0, 1], [0, 1]);

  // Glow pulse
  const glowPulse = Math.sin(frame * 0.08) * 0.3 + 0.7;

  // Subtitle
  const subReveal = sp(50);

  // Particles
  const particles = Array.from({ length: 40 }, (_, i) => ({
    x: seededRandom(i * 3) * 1920,
    y: seededRandom(i * 7) * 1080,
    size: seededRandom(i * 11) * 4 + 1,
    speed: seededRandom(i * 13) * 0.5 + 0.2,
    delay: seededRandom(i * 17) * 100,
  }));

  // Shape layers
  const shapes = Array.from({ length: 6 }, (_, i) => ({
    x: seededRandom(i * 31) * 1920,
    y: seededRandom(i * 37) * 1080,
    size: seededRandom(i * 41) * 300 + 100,
    rotation: seededRandom(i * 43) * 360,
    type: i % 3,
  }));

  // Rupee symbols floating
  const rupeeSymbols = Array.from({ length: 12 }, (_, i) => ({
    x: seededRandom(i * 53) * 1920,
    y: seededRandom(i * 59) * 1080,
    size: seededRandom(i * 61) * 40 + 20,
    speed: seededRandom(i * 67) * 0.3 + 0.1,
    opacity: seededRandom(i * 71) * 0.15 + 0.05,
  }));

  return (
    <AbsoluteFill
      style={{
        backgroundColor: '#1C1C1C',
        overflow: 'hidden',
        opacity: exitOp,
      }}
    >
      {/* Animated gradient background */}
      <AbsoluteFill
        style={{
          transform: `scale(${cameraZoom})`,
          background: `
            radial-gradient(ellipse at 50% 40%, rgba(255,193,7,0.15) 0%, transparent 60%),
            radial-gradient(ellipse at 80% 80%, rgba(233,215,0,0.08) 0%, transparent 50%),
            radial-gradient(ellipse at 20% 90%, rgba(255,0,0,0.06) 0%, transparent 40%),
            linear-gradient(180deg, #1C1C1C 0%, #0a0a0a 100%)
          `,
        }}
      />

      {/* Shape layers */}
      {shapes.map((s, i) => (
        <div
          key={`shape-${i}`}
          style={{
            position: 'absolute',
            left: s.x,
            top: s.y,
            width: s.size,
            height: s.size,
            borderRadius: s.type === 0 ? '50%' : s.type === 1 ? '0%' : '50% 0% 50% 0%',
            border: `1px solid rgba(255,193,7,${0.04 + Math.sin(frame * 0.02 + i) * 0.02})`,
            transform: `rotate(${s.rotation + frame * 0.15 * (i % 2 === 0 ? 1 : -1)}deg) scale(${1 + Math.sin(frame * 0.015 + i) * 0.05})`,
          }}
        />
      ))}

      {/* Motion texture - concentric rings */}
      <AbsoluteFill style={{ opacity: 0.04 }}>
        {Array.from({ length: 8 }, (_, i) => (
          <div
            key={`ring-${i}`}
            style={{
              position: 'absolute',
              left: '50%',
              top: '50%',
              width: 200 + i * 150,
              height: 200 + i * 150,
              borderRadius: '50%',
              border: '1px solid #FFC107',
              transform: `translate(-50%, -50%) scale(${1 + Math.sin(frame * 0.02 + i * 0.5) * 0.1})`,
              opacity: interpolate(frame, [0, 30], [0, 1], { extrapolateRight: 'clamp' }) * (1 - i * 0.1),
            }}
          />
        ))}
      </AbsoluteFill>

      {/* Floating ₹ symbols */}
      {rupeeSymbols.map((r, i) => (
        <div
          key={`rupee-${i}`}
          style={{
            position: 'absolute',
            left: r.x,
            top: r.y + Math.sin(frame * r.speed + i) * 30 - frame * 0.3,
            fontSize: r.size,
            fontFamily: 'Inter',
            fontWeight: 700,
            color: '#FFC107',
            opacity: r.opacity * interpolate(frame, [0, 40], [0, 1], { extrapolateRight: 'clamp' }),
            transform: `rotate(${Math.sin(frame * 0.03 + i) * 15}deg)`,
          }}
        >
          ₹
        </div>
      ))}

      {/* Particles */}
      {particles.map((p, i) => (
        <div
          key={`particle-${i}`}
          style={{
            position: 'absolute',
            left: p.x,
            top: (p.y + frame * p.speed * 2) % 1080,
            width: p.size,
            height: p.size,
            borderRadius: '50%',
            backgroundColor: i % 3 === 0 ? '#FFC107' : i % 3 === 1 ? '#e9d700' : '#F5F5F5',
            opacity: 0.3 + Math.sin(frame * 0.05 + p.delay) * 0.2,
          }}
        />
      ))}

      {/* Main content - Giant ₹1 CRORE */}
      <AbsoluteFill
        style={{
          display: 'flex',
          flexDirection: 'column',
          alignItems: 'center',
          justifyContent: 'center',
        }}
      >
        {/* Glow behind number */}
        <div
          style={{
            position: 'absolute',
            width: 600,
            height: 600,
            borderRadius: '50%',
            background: `radial-gradient(circle, rgba(255,193,7,${0.25 * glowPulse}) 0%, transparent 70%)`,
            transform: `scale(${numberReveal * 1.2})`,
          }}
        />

        {/* ₹1 Crore text */}
        <div
          style={{
            fontSize: 220,
            fontFamily: "'Yantra One'",
            fontWeight: 900,
            color: '#FFC107',
            opacity: numberOp,
            transform: `scale(${numberScale})`,
            textShadow: `0 0 60px rgba(255,193,7,${0.4 * glowPulse}), 0 0 120px rgba(255,193,7,0.15)`,
            letterSpacing: -5,
            lineHeight: 1,
          }}
        >
          ₹1 करोड़
        </div>

        {/* Subtitle */}
        <div
          style={{
            marginTop: 30,
            fontSize: 52,
            fontFamily: "'Noto Sans Devanagari'",
            fontWeight: 600,
            color: '#F5F5F5',
            opacity: interpolate(subReveal, [0, 1], [0, 0.9]),
            transform: `translateY(${interpolate(subReveal, [0, 1], [30, 0])}px)`,
          }}
        >
          वो नंबर जो सुनने में बड़ा लगता है
        </div>

        {/* Underline accent */}
        <div
          style={{
            marginTop: 15,
            width: interpolate(sp(65), [0, 1], [0, 400]),
            height: 3,
            background: 'linear-gradient(90deg, transparent, #FFC107, transparent)',
            borderRadius: 2,
          }}
        />
      </AbsoluteFill>

      {/* Stick figure - amazed pose */}
      <div
        style={{
          position: 'absolute',
          right: 120,
          bottom: 100,
          opacity: interpolate(sp(40), [0, 1], [0, 1]),
          transform: `translateY(${interpolate(sp(40), [0, 1], [60, 0])}px) translateY(${Math.sin(frame * 0.05) * 2}px)`,
        }}
      >
        <svg width="120" height="200" viewBox="0 0 120 200">
          {/* Glow */}
          <circle cx="60" cy="40" r="50" fill="rgba(255,193,7,0.08)" />
          {/* Head */}
          <circle cx="60" cy="40" r="20" fill="none" stroke="#F5F5F5" strokeWidth="3" transform={`rotate(${Math.sin(frame * 0.04) * 3}, 60, 40)`} />
          {/* Golden glasses */}
          <rect x="45" y="33" width="13" height="10" rx="2" fill="none" stroke="#FFC107" strokeWidth="2" />
          <rect x="62" y="33" width="13" height="10" rx="2" fill="none" stroke="#FFC107" strokeWidth="2" />
          <line x1="58" y1="38" x2="62" y2="38" stroke="#FFC107" strokeWidth="1.5" />
          {/* Mouth - O shape amazed */}
          <ellipse cx="60" cy="50" rx="4" ry="5" fill="none" stroke="#F5F5F5" strokeWidth="1.5" />
          {/* Body */}
          <line x1="60" y1="60" x2="60" y2="130" stroke="#F5F5F5" strokeWidth="3" />
          {/* Arms raised up in amazement */}
          <line x1="60" y1="85" x2="30" y2={60 + Math.sin(frame * 0.06) * 5} stroke="#F5F5F5" strokeWidth="3" />
          <line x1="60" y1="85" x2="90" y2={60 + Math.sin(frame * 0.06 + 1) * 5} stroke="#F5F5F5" strokeWidth="3" />
          {/* Legs */}
          <line x1="60" y1="130" x2="40" y2="180" stroke="#F5F5F5" strokeWidth="3" />
          <line x1="60" y1="130" x2="80" y2="180" stroke="#F5F5F5" strokeWidth="3" />
