import React from 'react';
import { useCurrentFrame, useVideoConfig, interpolate, Easing, spring, AbsoluteFill } from 'remotion';

export const compositionConfig = {
  id: 'Scene1Ch5',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Scene1Ch5: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Paper crash entrance
  const paperY = interpolate(sp(0), [0, 1], [-400, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const paperRot = interpolate(sp(0), [0, 1], [-8, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Pen draw animation
  const penProgress = interpolate(frame, [20, 80], [0, 1], { easing: Easing.out(Easing.cubic), extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Headline
  const headY = interpolate(sp(15), [0, 1], [60, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const headOp = interpolate(frame, [15, 40], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Sub line
  const subOp = interpolate(frame, [45, 70], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const subX = interpolate(sp(45), [0, 1], [-80, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Stamp STEP 1
  const stampScale = sp(60, { damping: 8, stiffness: 200, mass: 0.3 });
  const stampRot = interpolate(stampScale, [0, 1], [-15, -6], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Background pulse
  const bgPulse = interpolate(frame, [0, 180], [0, 1], { easing: Easing.inOut(Easing.sine), extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Stick figure breath
  const breath = Math.sin(frame * 0.05) * 2;
  const headNod = Math.sin(frame * 0.04) * 3;
  const figureOp = interpolate(frame, [50, 75], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Floating particles
  const particles = Array.from({ length: 18 }, (_, i) => ({
    x: seededRandom(i * 31) * 1920,
    y: seededRandom(i * 17) * 1080,
    r: seededRandom(i * 53) * 5 + 2,
    speed: seededRandom(i * 7) * 0.4 + 0.1,
    phase: seededRandom(i * 43) * Math.PI * 2,
  }));

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>

      {/* === GRADIENT BACKGROUND === */}
      <AbsoluteFill>
        <div style={{
          width: '100%', height: '100%',
          background: `radial-gradient(ellipse 120% 80% at ${30 + bgPulse * 10}% ${40 + bgPulse * 5}%, #2a2200 0%, #1C1C1C 55%, #0d1a0d 100%)`,
        }} />
      </AbsoluteFill>

      {/* === SHAPE LAYERS — Large drifting geometry === */}
      <AbsoluteFill style={{ opacity: 0.05 }}>
        {/* Big circle top-left */}
        <div style={{
          position: 'absolute', top: -120, left: -80,
          width: 520, height: 520,
          borderRadius: '50%',
          border: '3px solid #e9d700',
          transform: `rotate(${frame * 0.1}deg) scale(${1 + Math.sin(frame * 0.02) * 0.04})`,
        }} />
        {/* Rectangle center-right */}
        <div style={{
          position: 'absolute', top: 300, right: -100,
          width: 380, height: 380,
          border: '2px solid #FFC107',
          transform: `rotate(${25 + frame * 0.08}deg)`,
        }} />
        {/* Triangle-ish shape bottom */}
        <svg style={{ position: 'absolute', bottom: 0, left: 200 }} width="400" height="300">
          <polygon points="200,0 0,300 400,300"
            fill="none" stroke="#e9d700" strokeWidth="2"
            transform={`rotate(${frame * 0.06}, 200, 150)`} />
        </svg>
      </AbsoluteFill>

      {/* === MOTION TEXTURE — Grid lines === */}
      <AbsoluteFill style={{ opacity: 0.04 }}>
        <svg width="1920" height="1080">
          {Array.from({ length: 20 }, (_, i) => (
            <line key={`h${i}`} x1="0" y1={i * 57} x2="1920" y2={i * 57}
              stroke="#e9d700" strokeWidth="0.8" />
          ))}
          {Array.from({ length: 34 }, (_, i) => (
            <line key={`v${i}`} x1={i * 57} y1="0" x2={i * 57} y2="1080"
              stroke="#e9d700" strokeWidth="0.8" />
          ))}
        </svg>
      </AbsoluteFill>

      {/* === FLOATING PARTICLES === */}
      <AbsoluteFill style={{ pointerEvents: 'none' }}>
        {particles.map((p, i) => (
          <div key={i} style={{
            position: 'absolute',
            left: p.x,
            top: p.y + Math.sin(frame * p.speed + p.phase) * 18,
            width: p.r * 2, height: p.r * 2,
            borderRadius: '50%',
            backgroundColor: i % 3 === 0 ? '#e9d700' : i % 3 === 1 ? '#FFC107' : '#F5F5F5',
            opacity: 0.12 + Math.sin(frame * 0.03 + p.phase) * 0.06,
          }} />
        ))}
      </AbsoluteFill>

      {/* === MAIN PAPER ELEMENT === */}
      <AbsoluteFill style={{ display: 'flex', alignItems: 'center', justifyContent: 'flex-start', paddingLeft: 130 }}>
        <div style={{

            position: 'absolute', inset: 0,

            borderRadius: 12,
            transform: 'translate(12px, 12px)',
            opacity: 0.5,
          }} />
          {/* Paper body */}
          <div style={{
            position: 'absolute', inset: 0,
            backgroundColor: '#F5F5F5',
            borderRadius: 12,
            padding: '40px 48px',
          }}>
            {/* Paper lines */}
            {Array.from({ length: 8 }, (_, i) => {
              const lineWidth = interpolate(frame, [25 + i * 8, 55 + i * 8], [0, 100], {
                easing: Easing.out(Easing.cubic),
                extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
              });
              return (
                <div key={i} style={{
                  height: 3,
                  backgroundColor: i === 0 ? '#e9d700' : '#9E9E9E',
                  borderRadius: 2,
                  marginBottom: 34,
                  width: `${lineWidth * (i === 0 ? 0.85 : seededRandom(i * 11) * 0.4 + 0.45)}%`,
                  opacity: i === 0 ? 1 : 0.4,
                }} />
              );
            })}
            {/* Top label on paper */}
            <div style={{
              position: 'absolute', top: 20, left: 48,
              fontFamily: 'Noto Sans Devanagari, sans-serif',
              fontSize: 18, fontWeight: 800,
              color: '#1C1C1C',

            }}>मेरे निवेश — सूची</div>
          </div>

          {/* Pen drawing SVG overlay */}

            width="90" height="200" viewBox="0 0 90 200">
            <line x1="45" y1="20" x2="45" y2="180"
