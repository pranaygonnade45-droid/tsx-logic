import React from 'react';

  id: 'Scene1_Ch5_NahiKarunga',
  durationInSeconds: 12,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Strike-through animation on "शायद सुरक्षित"
  const strikeWidth = interpolate(frame, [40, 70], [0, 100], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });

  // Reveal lines stagger
  const line1Op = interpolate(frame, [10, 25], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const line2Op = interpolate(frame, [30, 45], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const line3Op = interpolate(frame, [55, 75], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Shake on strike
  const shakeX = interpolate(frame, [68, 70, 72, 74, 76], [0, -6, 6, -4, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Background pulse
  const bgPulse = interpolate(frame, [0, 60, 120], [0, 1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Floating particles
  const particles = Array.from({ length: 18 }, (_, i) => ({
    x: seededRandom(i * 3 + 1) * 1920,
    y: seededRandom(i * 3 + 2) * 1080,
    r: seededRandom(i * 3 + 3) * 6 + 2,
    speed: seededRandom(i + 77) * 0.8 + 0.3,

        }} />
        {/* Circle drifting */}
        <div style={{

          width: 420, height: 420,
          borderRadius: '50%',
          border: '1.5px solid rgba(255,0,0,0.05)',
